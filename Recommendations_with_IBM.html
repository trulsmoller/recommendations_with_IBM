def create_user_item_matrix(df):
    '''
    INPUT:
    df - pandas dataframe with article_id, title, user_id columns
    
    OUTPUT:
    user_item - pandas dataframe user item matrix 
    
    Description:
    Return a matrix with user ids as rows and article ids on the columns with 1 values where a user interacted with 
    an article and a 0 otherwise
    '''
    
    # initializing a the user-item matrix as a numpy array with zeros only
    n = len(df.user_id.drop_duplicates())
    k = len(df.article_id.drop_duplicates())
    user_item = np.zeros((n,k))
    
    # extracting the user_ids and the article ids (lists of strings) 
    user_ids = df.user_id.drop_duplicates().tolist()
    article_ids = df.article_id.drop_duplicates().tolist()
    
    # iterating through each row of df,
    # with the values for user_id and article_id present on that row,
    # using the position of those values in the user_ids list and article_ids list 
    # respectively as x/y "coordinates",
    # and putting a 1 in that coordinate place of the user-item matrix
    for index, row in df.iterrows():
        
        x = user_ids.index(row['user_id'])
        y = article_ids.index(row['article_id'])
        
        user_item[x, y] = 1
    
    # converting the user_item numpy array to a DataFrame and adding ids to index and column headers
    user_item = pd.DataFrame(user_item, index=user_ids, columns=article_ids)
    
    return user_item

def find_similar_users(user_id, user_item=user_item):
    '''
    INPUT:
    user_id - (int) a user_id
    user_item - (pandas dataframe) matrix of users by articles: 
                1's when a user has interacted with an article, 0 otherwise
    
    OUTPUT:
    most_similar_users - (list) an ordered list where the closest users (largest dot product users)
                    are listed first
    
    Description:
    Computes the similarity of every pair of users based on the dot product
    Returns an ordered list of most similar users. Order of ties are randomized.
    
    '''
    # compute similarity of each user to the provided user based using numpy dot product of arrays 
    # to sum up the number of matching articles in the user-item matrix
    # note: even the similarity between the provided user and itself is included at this point
    similarity_score_list = []
    n = user_item.shape[0]
    
    for i in range(n):
        matches = np.dot(user_item[user_item.index == user_id].values, user_item[user_item.index == i].T.values)[0]
        similarity_score_list.append(matches)

    # create a pandas Series from the similarity_score_list
    similarity_score = pd.Series(similarity_score_list)
    
    # sort Series by similarity (taking a full sample before sorting to achieve randomized order of ties)
    similarity_score_ordered = similarity_score.sample(frac=1, random_state=42).sort_values(ascending=False)
    
    # extract the most similar users' ids from the index of the Series
    most_similar_users = similarity_score_ordered.index.tolist()
    
    # remove the provided user from most_similar_users
    index_self = most_similar_users.index(user_id)
    most_similar_users = np.delete(most_similar_users, obj=index_self, axis=0).tolist()
    
    # return a list of the users in order from most to least similar
    
    return most_similar_users

def user_user_recs(user_id, m=10):
    '''
    INPUT:
    user_id - (int) a user id, later referred to as "the provided user"
    m - (int) the number of recommendations you want for the user
    
    OUTPUT:
    recs - (list) a list of recommendations (article ids) for the provided user
    
    Description:
    Loops through the users based on closeness to the input user_id
    For each user - finds articles the user hasn't seen before and accumulates them as recommendations
    Does this until m recommendations are found
    
    Notes:
    Users who are the same closeness are chosen arbitrarily as the 'next' user
    
    For the user where the number of recommended articles starts below m 
    and ends exceeding m, the last items are chosen arbitrarily
    
    '''
    # getting a list of the closest users
    users_by_closeness = find_similar_users(user_id)
    
    # initiating list for recommendations
    recs = [] 
    
    # getting the article id for the provided user id
    article_ids_provided_user, article_names_provided_user = get_user_articles(user_id)
    
    for ind, item in enumerate(users_by_closeness):
        
        try:
            # using this function to get article_ids for each neighbor (item) in the loop
            article_ids, article_names = get_user_articles(item)

            # list of unique article_ids that are not already seen by the provided user
            list_difference = list(set(article_ids) - set(article_ids_provided_user))

            # list of potential recommendations; list_differences that are not already in recs
            new_unique = list(set(list_difference) - set(recs))
            
            # calculating n as recommendations plus potential recommendations
            a = len(recs)
            b = len(new_unique)
            
            n = a + b
                    
            # in case n exceeds m, reduce new_unique with the difference, then add to recs
            if n >= m:
                diff = n - m
            
                new_unique_to_add = new_unique[:-diff]
                recs.extend(new_unique_to_add)
            
                break
                
            else:
                recs.extend(new_unique)
                
        except:
            pass
    
    
    return recs

def get_top_sorted_users(user_id, df=df, user_item=user_item):
    '''
    INPUT:
    user_id - (int)
    df - (pandas dataframe) df as defined at the top of the notebook 
    user_item - (pandas dataframe) matrix of users by articles: 
            1's when a user has interacted with an article, 0 otherwise
    
            
    OUTPUT:
    neighbors_df - (pandas dataframe) a dataframe with:
                    neighbor_id - is a neighbor user_id
                    similarity - measure of the similarity of each user to the provided user_id
                    num_interactions - the number of articles viewed by the user - if a u
                    
    Other Details - sort the neighbors_df by the similarity and then by number of interactions where 
                    highest of each is higher in the dataframe. Ties are randomized.
     
    '''
    assert user_id > 0, "User_id must be positive"
    user_ind = user_id - 1
    # compute similarity of each user to the provided user based using numpy dot product of arrays 
    # to sum up the number of matching articles in the user-item matrix
    # note: even the similarity between the provided user and itself is included at this point
    similarity_score_list = []
    n = user_item.shape[0]
    for i in user_item.index:
        matches = np.dot(user_item[user_item.index == user_id].values.flatten(), user_item[user_item.index == i].values.flatten())
        
        similarity_score_list.append(matches)
        
    # create a pandas Series from the similarity_score_list
    similarity_score = pd.Series(similarity_score_list)
    similarity_score.index += 1
    
    
    # create pandas Series of frequency each article has been seen from the user-item
    frequency = pd.Series(user_item.sum(axis=1))
    
    
    # concat the two Series to DataFrame, add headers and sort by similarity, then num_interactions
    dfa = pd.concat([similarity_score, frequency], axis=1).reset_index()
    dfa.columns = ['neighbor_id','similarity', 'num_interactions']
    
    # doing a full sample before sorting to achive random order in case of tie in 
    # both similarity AND num_interactions
    dfa = dfa.sample(frac=1, random_state=42).sort_values(by=['similarity', 'num_interactions'], 
                                                          ascending=False).reset_index(drop=True)
    
    # exclude the subject user_id
    neighbors_df = dfa[dfa.neighbor_id != user_id].reset_index(drop=True)
    
    
    
    return neighbors_df

latent_feats = np.arange(10, 700 + 10, 20)

for k in latent_feats:
    # restructure with k latent features
    s_train_lat, u_train_lat, vt_train_lat = np.diag(s_train[:k]), u_train[:, :k], vt_train[:k, :]
    
    u_test_lat, vt_test_lat = u_test[:, :k], vt_test[:k, :]
    
    # take dot product
    user_item_train_preds = np.around(np.dot(np.dot(u_train_lat, s_train_lat), vt_train_lat))
    user_item_test_preds = np.around(np.dot(np.dot(u_test_lat, s_train_lat), vt_test_lat))
    
    
    # compute prediction accuracy
    train_errors.append(accuracy_score(user_item_train.values.flatten(), user_item_train_preds.flatten()))
